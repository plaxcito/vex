def build_trajectory_graph(embeddings, k=5):
    G = nx.Graph()
    # Temporal edges
    for i in range(len(embeddings)-1):
        G.add_edge(i, i+1, weight=temporal_weight, type='temporal')

def compute_forman_ricci(G):
